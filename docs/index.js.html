        options.endb = this;
        options.name = typeof name === 'string' ? name.replace(/[^a-z0-9]/gi, '_').toLowerCase() : 'endb'; 
        options.columns = Object.keys(schema).map(col => (`\`${col}\` ${schema[col]}`));

        this.models = [];
        this.models.push(model);
        return model;
